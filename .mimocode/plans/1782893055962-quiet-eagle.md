# Plan: Cursor Pagination & Dynamic Pagination

## Overview

Add two new pagination strategies to the existing offset-based system:
1. **CursorPagination** — keyset/cursor-based using base64-encoded column value
2. **DynamicPagination** — auto-selects offset vs cursor based on total row count

Both are class instances, configured when creating a ModelAdmin. Default = `OffsetPagination()`.

---

## Design Decisions

| Decision | Choice |
|---|---|
| Default pagination | `OffsetPagination()` (unchanged behavior) |
| Configuration style | Class instances: `pagination = CursorPagination(cursor_column="created_at")` |
| Cursor column | Configurable via constructor, default = primary key |
| Dynamic threshold | Configurable via constructor, default = 1000 rows |
| Cursor format | Base64-encoded opaque string (`MjA=` for id=20) |

---

## Usage Example

```python
from fastapi_console.pagination import CursorPagination, DynamicPagination, OffsetPagination

# Default (same as before)
class UserAdmin(ModelAdmin):
    per_page = 20

# Cursor pagination on created_at column
class LogAdmin(ModelAdmin):
    pagination = CursorPagination(cursor_column="created_at")
    per_page = 50

# Dynamic: offset for small tables, cursor for large (threshold=500)
class EventAdmin(ModelAdmin):
    pagination = DynamicPagination(cursor_column="id", threshold=500)
    per_page = 25
```

### 4. `fastapi_console/pagination/cursor.py`
```python
class CursorPagination(BasePagination):
    def __init__(self, cursor_column: str | None = None):
        self.cursor_column = cursor_column  # None = use PK

    def _decode_cursor(self, cursor: str) -> Any:
        return json.loads(base64.b64decode(cursor))

    def _encode_cursor(self, value: Any) -> str:
        return base64.b64encode(json.dumps(value).encode()).decode()

    async def paginate(self, stmt, session, per_page, page=1, after=None, before=None, pk_col=None, model=None) -> PaginationResult:
        # Determine cursor column (use PK if not set)
        col = getattr(model, self.cursor_column) if self.cursor_column else pk_col

        if after:
            cursor_val = self._decode_cursor(after)
            stmt = stmt.where(col > cursor_val)
        elif before:
            cursor_val = self._decode_cursor(before)
            stmt = stmt.where(col < cursor_val)
            stmt = stmt.order_by(col.desc())  # reverse for backward

        # Count total
        count_q = select(func.count()).select_from(stmt.subquery())
        total = (await session.execute(count_q)).scalar() or 0

        # Fetch per_page + 1 to detect has_next
        stmt = stmt.limit(per_page + 1)
        result = await session.execute(stmt)
        items = list(result.unique().scalars().all())

        has_next = len(items) > per_page
        if has_next:
            items = items[:per_page]

        next_cursor = None
        if has_next and items:
            last_val = getattr(items[-1], self.cursor_column or "id")
            next_cursor = self._encode_cursor(last_val)

        return PaginationResult(
            items=items, total=total, per_page=per_page,
            next_cursor=next_cursor, has_next=has_next,
        )
```

### 5. `fastapi_console/pagination/dynamic.py`
```python
class DynamicPagination(BasePagination):
    def __init__(self, cursor_column: str | None = None, threshold: int = 1000):
        self.cursor_column = cursor_column
        self.threshold = threshold
        self._offset = OffsetPagination()
        self._cursor = CursorPagination(cursor_column=cursor_column)

    async def paginate(self, stmt, session, per_page, **kw) -> PaginationResult:
        # Count total first
        count_q = select(func.count()).select_from(stmt.subquery())
        total = (await session.execute(count_q)).scalar() or 0

        if total <= self.threshold:
            return await self._offset.paginate(stmt, session, per_page, **kw)
        else:
            return await self._cursor.paginate(stmt, session, per_page, **kw)
```
